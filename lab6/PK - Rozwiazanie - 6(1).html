function legendrePoly(n)
    P_0 = Polynomial([1])
    P_1 = Polynomial([0,1])
    if n == 0
        return P_0
    elseif n == 1
        return P_1
    end
    k = 1
    P_result = 0
    while k < n
        scalar1 = (2*k+1)/(k+1)
        scalar0 = k/(k+1)
        P_x = Polynomial([0,1])  # (p(x) = x)
        P_result = scalar1*P_x*P_1 - scalar0*P_0
        P_0 = P_1
        P_1 = P_result
        k += 1
        
    end
    return P_result
end

function compare_arrays(arr_a, arr_b)  #both arrays are presorted
    eps = 0.0001
    for i in 1:length(arr_a)
        if abs(arr_a[i] - arr_b[i]) > eps
            return false
        end
    end
    return true
end

function gaussian_exact_integral(f, k, a, b)
    (xp, w) = gauss(Float64,k)
    diff = (b-a)/2
    avg = (a+b)/2
    diff*sum(w .* f.(diff.*xp .+ avg))
end

function rectangular_integral(f, n, a, b)   #f - function, n - number of devisions, a, b - range
    h = (b-a)/n
    xi = (a+0.5*h):h:(b-0.5*h)
    yi = [f(x) for x in xi]
    sum([y*h for y in yi])
end
